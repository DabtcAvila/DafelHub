VaultManager.ts

AES-256-GCM
/**
 * Enterprise Vault Manager
 * AES-256-GCM encryption with key rotation
 */
export class VaultManager {
  private readonly config: EncryptionConfig;
  private masterKey: Buffer;
  private keyVersion: number = 1;

  public async encrypt(plaintext: string): Promise<string> {
    const salt = crypto.randomBytes(this.config.saltLength);
    const key = await this.deriveKey(this.masterKey, salt);
    const iv = crypto.randomBytes(this.config.ivLength);
    
    const cipher = crypto.createCipheriv('aes-256-gcm', key, iv);
    const encrypted = Buffer.concat([
      cipher.update(plaintext, 'utf8'),
      cipher.final(),
    ]);
    
    const tag = cipher.getAuthTag();
    return this.encodeSecurely(encrypted, iv, tag, salt);
  }
}

PostgreSQLConnector.ts

Enterprise
/**
 * Enterprise PostgreSQL Connector
 * Production-ready with connection pooling
 */
export class PostgreSQLConnector implements ISQLConnector {
  private pool?: Pool;
  private metrics: MetricsCollector;
  private activeQueries: Set<string> = new Set();

  public async query<T>(query: string, params?: any[]): Promise<QueryResult<T>> {
    const queryId = this.generateQueryId();
    const startTime = Date.now();
    
    this.activeQueries.add(queryId);
    this.emit('query_start', { queryId, query });
    
    try {
      const result = await this.pool.query<T>(query, params);
      const duration = Date.now() - startTime;
      
      this.updateMetrics(true, duration);
      return this.formatResult(result, duration);
    } finally {
      this.activeQueries.delete(queryId);
    }
  }
}

Data Sources API

Next.js
/**
 * POST /api/data-sources
 * Create new data source with encryption
 */
export async function POST(request: Request) {
  const vault = VaultManager.getInstance();
  const connectionManager = ConnectionManager.getInstance();
  
  const { name, type, config } = await request.json();
  
  // Encrypt sensitive credentials
  const encryptedPassword = await vault.encrypt(config.password);
  const encryptedApiKey = config.apiKey 
    ? await vault.encrypt(config.apiKey) 
    : null;

  // Save to database with encryption
  const dataSource = await prisma.dataSource.create({
    data: {
      name,
      type,
      password: encryptedPassword,
      apiKey: encryptedApiKey,
      createdById: session.user.id,
    },
  });

  return NextResponse.json(dataSource);
}
